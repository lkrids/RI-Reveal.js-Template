Best Practices

Ride the magical stead.

Use A CDN

							
<script type="text/javascript" src="http://ajax.googleapis.com/ajax/libs/jquery/1.7.2/jquery.min.js"></script>


Use development version for development



Use the latest version



Put scripts at the bottom of page to prevent blocking

selector performance

							
var notSoFast = $('[attr="this"]'),
    fast = $('.class'),
    fastest = $('#id');

Cache objects

							
var $obj = $('#obj'),
    $sub = $obj.find('#sub');

Use chaining

							
// Bad
$('#id').detach();
$('#id').css({
      'top': 20,
      'opacity': 0.8
    });
$('#id').appendTo($something);

// Good
$('#id').detach()
    .css({
      'top': 20,
      'opacity': 0.8
    })
    .appendTo($something);

Use .on() for event delegation

							
// Bad
$('#id').click( ... );

// Good
$('#id').on('click', ... );

Minimize DOM Touches

							
// Remove obj from DOM when doing work.
$('#id').detach().css( ... ).appendTo( ... );

// Utilize documentFragment
var $hook = $('#id'),
    frag = document.createDocumentFragment(),
    element,
    i;

for (i = 0; i < 5; i   ) {
    element  = '<span>Element</span>';
}

frag.append(element);
$hook.append(frag);
